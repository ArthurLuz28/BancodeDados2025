CREATE TABLE proprietario (
    codpro int PRIMARY KEY,
    nome varchar(255),
    datacadastro date,
    email varchar(255),
    telefone varchar(25),
    cpf varchar(15),
    fk_corretor_codcorr int
);

CREATE TABLE corretor (
    codcorr int PRIMARY KEY,
    nome varchar(255),
    CRECI varchar(20),
    telefone varchar(25),
    experiencia varchar(255),
    cpf varchar(15),
    fk_endereco_codendereco int
);

CREATE TABLE imovel (
    codimovel int PRIMARY KEY,
    qtdquartos int,
    qtdgaragem int,
    qtdbanheiros int,
    valoraluguel float,
    ano_construcao date,
    piscina bit,
    qtdcameras_seguranca int,
    fk_proprietario_codpro int,
    fk_endereco_codendereco int
);

CREATE TABLE inquilino (
    codinq int PRIMARY KEY,
    rg varchar(15),
    cpf varchar(15),

    nome varchar(255),

    telefone varchar(25),
    fk_corretor_codcorr int
);

CREATE TABLE endereco (
    codendereco int PRIMARY KEY,
    rua varchar(255),
    bairro varchar(255),
    cep varchar(255),
    complemento varchar(255),
    num varchar(10),
    uf char(2),
    cidade varchar(255),
    pais varchar(255),
    CEP varchar(9)
);

CREATE TABLE aluga (
